Select Distinct
       dm.User_Id,
       d.dep_Code,
			 dm.dep_Mmbr_Pstn_Id,
       dmps.Position_Name,
			 Stp.Pkg_User.F_Get_User_Fio(dm.User_Id) User_Name,
       Stp.Pkg_User.F_Get_User_Prof(dm.User_Id) User_Prof
From   Stp.v_Dep_Mmbrs dm
Left   Join Stp.v_Dep_Mmbr_Pstns dmps
On     dmps.Id_dep_Mmbr_Pstn=dm.dep_Mmbr_Pstn_Id
Left   Join Stp.v_Deps d
On     d.id_dep=dm.dep_Id
       And Nvl(d.Date_Expitation, Sysdate)<=Sysdate
Where  dm.dep_Mmbr_Pstn_Id=201
Order  By d.dep_Code
