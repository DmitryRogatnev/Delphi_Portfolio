---------------------
/*BeginColumns
Coeff_Flag_Name
EndColumns*/

/*BeginTitles
Коэффициент
EndTitles*/
--------------------
Select ( Case Rownum 
           When 1 Then 'unit_time'
           When 2 Then 'oper_time'
         End ) Coeff_Flag, 
       ( Case Rownum 
           When 1 then '% от Т шт.'
           When 2 then '% от Т оп.'
         End ) Coeff_Flag_Name
From   Dual
Where  /*BeginFilter*/ (1=1) /*EndFilter*/
Connect By Level < 3
